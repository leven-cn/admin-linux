Usage


     import os
     import admin
     configs = {'PROJECT_NAME': '"AAA"',
           'PROJECT_NUMBER': '1.2.3'}
     try:
         with open(os.path.join(os.getcwd(), 'Doxyfile'), 'r+') as f:
             config_file = admin.ConfigFile(f)
             config_file.set(configs)
         except IOError as e:
             admin.error('error')
