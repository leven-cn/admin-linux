127  def serve_forever(self, multiplex=None, timeout=0.5):
128  '''Run server.
129 
130  @param multiplex I/O multiplexing method: "None", "epoll", "poll", "select".
131  @param timeout timeout used by POSIX `select()` or poll()` in seconds
132  @exception select.error
133 
134  The `poll()` system call, supported on most Unix systems, provides
135  better scalability for network servers that service many, many clients
136  at the same time. `poll()` scales better because the system call only
137  requires listing the file descriptors of interest, while `select()`
138  builds a bitmap, turns on bits for the fds of interest, and then
139  afterwards the whole bitmap has to be linearly scanned again. `select()`
140  is O(highest file descriptor), while poll() is O(number of file
141  descriptors).
142 
143  @warnning `epoll()` ONLY available on Linux 2.5+.
144  '''
145  poller = None # epoll() or poll()
146  if multiplex == 'epoll':
147  # Setup epoller
148  poller = select.epoll()
149 
150  # epoll() event mask
151  #
152  # Constant | Meaning
153  # -------------------------
154  # EPOLLIN | Available for read
155  # EPOLLOUT | Available for write
156  # EPOLLPRI | Urgent data for read
157  # EPOLLERR | Error condition happened on the associated fd
158  # EPOLLHUP | Hang up happened on the associated fd
159  # EPOLLET | Set Edge Trigger behavior, the default is Level Trigger behavior
160  # EPOLLONESHOT | Set one-shot behavior. After one event is pulled out, the fd is internally disabled
161  # EPOLLRDNORM | Equivalent to EPOLLIN
162  # EPOLLRDBAND | Priority data band can be read.
163  # EPOLLWRNORM | Equivalent to EPOLLOUT
164  # EPOLLWRBAND | Priority data may be written.
165  # EPOLLMSG | Ignored.
166  poller.register(self.socket.fileno(), select.EPOLLIN)
167 
168  elif multiplex == 'poll':
169  # `timeout` value is represented in milliseconds,
170  # instead of seconds.
171  timeout *= 1000
172 
173  # Set event mask
174  #
175  # Constant | Meaning
176  # -------------------------
177  # POLLIN | There is data to read
178  # POLLPRI | There is urgent data to read
179  # POLLOUT | Ready for output: writing will not block
180  # POLLERR | Error condition of some sort
181  # POLLHUP | Hung up
182  # POLLNVAL | Invalid request: descriptor not open
183  READ_ONLY = (select.POLLIN|select.POLLPRI|select.POLLHUP|select.POLLERR)
184  READ_WRITE = (READ_ONLY|select.POLLOUT)
185 
186  # Setup poller
187  poller = select.poll()
188  poller.register(self.socket, READ_ONLY)
189 
190  # Map file descriptors to socket objects
191  fd_to_sockobj = {self.socket.fileno(): self.socket}
192 
193  while True:
194  # Wait for client request
195  print('Waiting for client on port {0}...'
196  .format(self.server_address[1]))
197 
198  # Set I/O multiplexing
199  if multiplex == 'epoll':
200  epoll_list = poller.poll()
201  for fd, events in epoll_list:
202  if fd == self.socket.fileno():
203  self._handle_request()
204 
205  # Polling (poll or select) reduces our responsiveness to a
206  # shutdown request and wastes CPU at all other times.
207  elif multiplex == 'poll':
208  events = poller.poll(timeout)
209  for fd, event in events:
210  s = fd_to_sockobj[fd]
211  if event & (select.POLLIN | select.POLLPRI):
212  if s == self.socket:
213  self._handle_request()
214 
215  elif multiplex == 'select':
216  r, w, e = _eintr_retry(select.select, [self.socket], [], [],
217  timeout)
218 
219  # Handle request without blocking
220  if self.socket in r:
221  self._handle_request()
222 
223  else:
224  self._handle_request()
225 
226  # Clean up
227  if multiplex in ['poll', 'epoll']:
228  poller.unregister(self.socket)
229  self.socket.close()
