99 #

101 # @param prefix prefix string of version
102 # @return Version named-tuple
103 def decode_version(version_info, prefix=''):
104  # Skip if it already decoded
105  if isinstance(version_info, Version):
106  return version_info

108  version_info.strip()
109  version_info = version_info[len(prefix):].strip().split('.')

117 ## Match the specific version.
118 #
119 # @param version Version named-tuple
120 # @param match version string (e.g. 1.2.3)
121 # @return True if match
122 def match_version(version, match):
123  version = decode_version(version)
124  match = decode_version(match)
125 
126  if version.major < match.major:
127  return False
128 
129  return (version.major > match.major) \
130  or (version.major == match.major and version.minor > match.minor) \
131  or (version.minor == match.minor and version.patch >= match.patch)

200  #
201  # @param pairs Pairs of option name-value.
202  def set(self, pairs):
203  self._config_file.seek(0)
204  lines = []
205  for line in self._config_file:
206  # Skip blank line or comments
